115: // Convert 2D array to CSV string
116: function convertToCSV(data) {
117:   return data.map(row => {
118:     return row.map(val => {
119:       let str = val === null ? '' : val.toString();
120:       if (str.indexOf(',') !== -1 || str.indexOf('"') !== -1 || str.indexOf('\n') !== -1) {
121:         str = '"' + str.replace(/"/g, '""') + '"';
122:       }
123:       return str;
124:     }).join(',');
125:   }).join('\n');
126: }
